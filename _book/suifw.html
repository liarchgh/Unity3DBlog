    //窗口从隐藏到显示时调用
    public override void Display()
    {
        base.Display();
    }

    //从Freeze状态到重新显示时调用
    public override void Redisplay()
    {
        base.Redisplay();
    }

    //窗体显示在其他窗体下面
    public override void Freeze()
    {
        base.Freeze();
    }

    //页面隐藏(不在“栈”集合中)
    public override void Hiding()
    {
        base.Hiding();
    }
    #endregion
